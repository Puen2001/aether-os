# precommit-scan — universal secret-key patterns (egrep).
# Public knowledge; safe to commit. User-specific private names go in
# system/denylist.txt (gitignored).
#
# Format: one egrep pattern per line. Lines starting with `#` ignored.
# Trailing `# comment` allowed on a pattern line.

sk-[A-Za-z0-9]{20,}                      # OpenAI / Anthropic (classic)
sk-ant-[A-Za-z0-9_-]{30,}                # Anthropic
sk-proj-[A-Za-z0-9_-]{30,}               # OpenAI project key
hf_[A-Za-z0-9]{30,}                      # Hugging Face
ghp_[A-Za-z0-9]{36}                      # GitHub Personal Access Token
gho_[A-Za-z0-9]{36}                      # GitHub OAuth
ghs_[A-Za-z0-9]{36}                      # GitHub server-to-server
ghu_[A-Za-z0-9]{36}                      # GitHub user-to-server
github_pat_[A-Za-z0-9_]{82}              # GitHub fine-grained PAT
xox[abprs]-[A-Za-z0-9-]{10,}             # Slack tokens
AKIA[0-9A-Z]{16}                         # AWS access key id
ASIA[0-9A-Z]{16}                         # AWS temporary key id
AIza[0-9A-Za-z_-]{35}                    # Google API key
sk_live_[A-Za-z0-9]{16,}                 # Stripe live secret key
rk_live_[A-Za-z0-9]{16,}                 # Stripe restricted live key
-----BEGIN [A-Z ]*PRIVATE KEY-----       # PEM private keys

# Generic / vendor-agnostic shapes — catch secrets that carry no known prefix
eyJ[A-Za-z0-9_=-]{8,}\.eyJ[A-Za-z0-9_=-]{8,}\.[A-Za-z0-9_=-]+   # JWT (header.payload.sig)
Bearer[[:space:]]+[A-Za-z0-9._~+/-]{20,}                        # Bearer token
[A-Za-z][A-Za-z0-9+.-]*://[^/:@[:space:]]+:[^/@[:space:]]+@     # credentials in a URL / connection string
AccountKey=[A-Za-z0-9+/]{40,}={0,2}                             # Azure storage account key
(api[_-]?key|secret|password|passwd|client_secret|access_token|auth_token)[[:space:]]*[:=][[:space:]]*["'][^"']{16,}   # secret in a quoted assignment
